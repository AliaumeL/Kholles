\exercice 

\begin{enumerate}
    \item Établir que pour $x \in ]0, +\infty[$ l'inégalité suivante
        est vérifiée
        \begin{equation*}
            \frac{1}{1+x} < \ln (1 + x) - \ln x < \frac{1}{x}
        \end{equation*}

    \item En déduire que pour $x \in ]0,+\infty[$ on a
        \begin{equation*}
            \left(1 + \frac{1}{x}\right)^x < e < \left(1 + \frac{1}{x}\right)^{1 + x}
        \end{equation*}
\end{enumerate}
