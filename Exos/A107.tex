\exercice["Equations différentielles"]

\begin{enumerate}
    \item Trouver une fonction dérivable $f$ telle que $f'(x) = f(x)$
    \item Trouver une fonction dérivable $f$ telle que 
        $f'(x) = - f (x)$
    \item Trouver une fonction dérivable $f$ telle que 
        $f'(x) = -x f(x)$
\end{enumerate}
