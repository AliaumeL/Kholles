\exercice[Exponentielle]

On pose 
\begin{equation*}
    I_n = \int_0^1 \frac{(1 - x)^n}{n!} e^x \di x
\end{equation*}

\begin{enumerate}[(a)]
    \item Montrer que la suite $I_n$ tend vers zéro
    \item Montrer que 
        \begin{equation*}
            I_n = \frac{1}{(n+1)!} + I_{n+1}
        \end{equation*}
    \item En déduire que 
        \begin{equation*}
            e = \sum_{k = 0}^\infty \frac{1}{k!}
        \end{equation*}
\end{enumerate}
